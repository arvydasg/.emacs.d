(setq user-init-file "C:/path/to/.emacs") ---- need this
(setq user-emacs-directory "C:/path/to/.emacs.d/") --- optional
(setq default-directory "C:/whatever/you/want/to/start/in") --- optional
(setenv "HOME" "D:/my/home/directory") --- optional
(load user-init-file) ---- need this
